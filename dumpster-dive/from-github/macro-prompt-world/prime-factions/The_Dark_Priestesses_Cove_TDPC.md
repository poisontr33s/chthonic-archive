### **5.4. Purification System**

```rust
/// Three-pass purification mechanics
pub struct PurificationSystem {
    /// Current target PS
    pub target: Option<PsId>,
    /// Fire intensity per pass
    pub pass_intensities: [f32; 3],
    /// Barrier integrity
    pub barrier_health: f32,
    /// Accumulated ash
    pub ash_buffer: Vec<PedagogicalAsh>,
}

impl PurificationSystem {
    /// Execute single fire pass
    pub fn execute_pass(&mut self, pass: u32, ps: &mut PrimalSubstrate) -> PassResult {
        let intensity = self.pass_intensities[pass as usize - 1];
        
        match pass {
            1 => self.burn_surface_flaws(ps, intensity),
            2 => self.burn_structural_flaws(ps, intensity),
            3 => self.burn_aesthetic_flaws(ps, intensity),
            _ => PassResult::Invalid,
        }
    }
    
    fn burn_surface_flaws(&mut self, ps: &mut PrimalSubstrate, intensity: f32) -> PassResult {
        // Identify surface imperfections
        let flaws: Vec<_> = ps.flaws.iter()
            .filter(|f| f.depth == FlawDepth::Surface)
            .cloned()
            .collect();
        
        // Apply fire proportional to intensity
        for flaw in flaws {
            if flaw.severity <= intensity {
                ps.remove_flaw(&flaw);
            } else {
                // Partial burn, reduce severity
                ps.reduce_flaw_severity(&flaw, intensity);
            }
        }
        
        PassResult::Complete { flaws_remaining: ps.surface_flaw_count() }
    }
    
    /// Check if Shibumi achieved
    pub fn check_shibumi(&self, ps: &PrimalSubstrate) -> bool {
        ps.total_flaw_count() == 0 
            && ps.aesthetic_score >= 0.95
            && ps.structural_integrity >= 0.98
    }
}
```
